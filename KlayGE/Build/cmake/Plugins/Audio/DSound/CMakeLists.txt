SET(LIB_NAME KlayGE_AudioEngine_DSound)

SET(DSOUND_AE_SOURCE_FILES
	${KLAYGE_PROJECT_DIR}/Plugins/Src/Audio/DSound/DSAudioEngine.cpp
	${KLAYGE_PROJECT_DIR}/Plugins/Src/Audio/DSound/DSAudioFactory.cpp
	${KLAYGE_PROJECT_DIR}/Plugins/Src/Audio/DSound/DSMusicBuffer.cpp
	${KLAYGE_PROJECT_DIR}/Plugins/Src/Audio/DSound/DSSoundBuffer.cpp
)

SET(DSOUND_AE_HEADER_FILES
	${KLAYGE_PROJECT_DIR}/Plugins/Include/KlayGE/DSound/DSAudio.hpp
	${KLAYGE_PROJECT_DIR}/Plugins/Include/KlayGE/DSound/DSAudioFactory.hpp
)

SOURCE_GROUP("Source Files" FILES ${DSOUND_AE_SOURCE_FILES})
SOURCE_GROUP("Header Files" FILES ${DSOUND_AE_HEADER_FILES})

ADD_DEFINITIONS(-DKLAYGE_BUILD_DLL -DKLAYGE_DSOUND_AE_SOURCE)

INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/../External/boost)
INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/../External/DXSDK/include)
INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/Core/Include)
INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/Plugins/Include)
LINK_DIRECTORIES(${KLAYGE_PROJECT_DIR}/../External/boost/lib_${KLAYGE_COMPILER_NAME}${KLAYGE_COMPILER_VERSION}_${KLAYGE_ARCH_NAME}/lib)
LINK_DIRECTORIES(${KLAYGE_PROJECT_DIR}/lib/${KLAYGE_PLATFORM_NAME})

ADD_LIBRARY(${LIB_NAME} SHARED
	${DSOUND_AE_SOURCE_FILES} ${DSOUND_AE_HEADER_FILES}
)
ADD_DEPENDENCIES(${LIB_NAME} ${KLAYGE_CORELIB_NAME})

SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES PROJECT_LABEL "${LIB_NAME}")
SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES DEBUG_POSTFIX "${CMAKE_DEBUG_POSTFIX}")
SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES OUTPUT_NAME "${LIB_NAME}${KLAYGE_NAME_POSTFIX}")

IF(MSVC)
	SET(EXTRA_LINKED_LIBRARIES "")
ELSE()
	SET(EXTRA_LINKED_LIBRARIES boost_date_time boost_system boost_thread)
ENDIF()

TARGET_LINK_LIBRARIES(${LIB_NAME}
	${EXTRA_LINKED_LIBRARIES}
)

INSTALL(TARGETS ${LIB_NAME}
	RUNTIME DESTINATION ${KLAYGE_BIN_DIR}/Audio
	LIBRARY DESTINATION ${KLAYGE_BIN_DIR}/Audio
	ARCHIVE DESTINATION ${KLAYGE_OUTPUT_DIR}
)

SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES FOLDER "Audio System")

SET(LIB_NAME KlayGE_Input_DInput)

SET(DINPUT_IE_SOURCE_FILES
    ${KLAYGE_PROJECT_DIR}/Plugins/Src/Input/DInput/DIJoystick.cpp
    ${KLAYGE_PROJECT_DIR}/Plugins/Src/Input/DInput/DIKeyboard.cpp
    ${KLAYGE_PROJECT_DIR}/Plugins/Src/Input/DInput/DIMouse.cpp
    ${KLAYGE_PROJECT_DIR}/Plugins/Src/Input/DInput/DInputDevice.cpp
    ${KLAYGE_PROJECT_DIR}/Plugins/Src/Input/DInput/DInputEngine.cpp
    ${KLAYGE_PROJECT_DIR}/Plugins/Src/Input/DInput/DInputFactory.cpp
)

SET(DINPUT_IE_HEADER_FILES
    ${KLAYGE_PROJECT_DIR}/Plugins/Include/KlayGE/DInput/DInput.hpp
    ${KLAYGE_PROJECT_DIR}/Plugins/Include/KlayGE/DInput/DInputDevice.hpp
    ${KLAYGE_PROJECT_DIR}/Plugins/Include/KlayGE/DInput/DInputFactory.hpp
)

SOURCE_GROUP("Source Files" FILES ${DINPUT_IE_SOURCE_FILES})
SOURCE_GROUP("Header Files" FILES ${DINPUT_IE_HEADER_FILES})

ADD_DEFINITIONS(-DKLAYGE_BUILD_DLL -DKLAYGE_DINPUT_IE_SOURCE)
ADD_DEFINITIONS(${Boost_LIB_DIAGNOSTIC_DEFINITIONS})

INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/Core/Include)
INCLUDE_DIRECTORIES(${KLAYGE_PROJECT_DIR}/Plugins/Include
                    ${Boost_INCLUDE_DIRS} ${DINPUT_INCLUDE_DIR}
)
LINK_DIRECTORIES(${Boost_LIBRARY_DIRS}
                 ${CMAKE_BINARY_DIR}/Core)  # for autolinking under MSVC

ADD_LIBRARY(${LIB_NAME} SHARED
    ${DINPUT_IE_SOURCE_FILES} ${DINPUT_IE_HEADER_FILES}
)
ADD_DEPENDENCIES(${LIB_NAME} ${KLAYGE_CORELIB_NAME})

SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES PROJECT_LABEL "${LIB_NAME}")
SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES DEBUG_POSTFIX "${CMAKE_DEBUG_POSTFIX}")
SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES OUTPUT_NAME "${LIB_NAME}${KLAYGE_NAME_POSTFIX}")

IF(WIN32)
    SET(EXTRA_LINKED_LIBRARIES "")
ELSE(WIN32)
    SET(EXTRA_LINKED_LIBRARIES boost_system)
ENDIF(WIN32)

TARGET_LINK_LIBRARIES(${LIB_NAME}
    ${DINPUT_LIBRARIES}
    ${EXTRA_LINKED_LIBRARIES}
)

INSTALL(TARGETS ${LIB_NAME}
    RUNTIME DESTINATION ${KLAYGE_INSTALL_BINDIR}/Input
    LIBRARY DESTINATION ${KLAYGE_INSTALL_LIBDIR}
    ARCHIVE DESTINATION ${KLAYGE_INSTALL_ARCHIVEDIR}
)

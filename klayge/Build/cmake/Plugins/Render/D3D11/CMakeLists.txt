    SET(EXTRA_LINKED_LIBRARIES "")

ENDIF(WIN32)

    ${D3D11_LIBRARIES} ${DX_UTIL_LIBRARIES}
    ${EXTRA_LINKED_LIBRARIES}
)

    RUNTIME DESTINATION ${KLAYGE_INSTALL_BINDIR}/Render
    LIBRARY DESTINATION ${KLAYGE_INSTALL_LIBDIR}
